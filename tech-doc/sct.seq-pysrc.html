 85      Parse fasta files. Adapted from: 
 86      http://www.petercollingridge.co.uk/python-bioinformatics-tools/fasta-parser 
 87   
 88      @type  filename: string 
 89      @param filename: Path to fasta file to be read. 
 90      @rtype:   list, dictionary 
 91      @return:  1. Sequence identifiers found in order they appear in the file 
 92   
 93                2. Dictionary where keys are sequence identifiers and values are 
 94                sequences as strings (single letter amino acid codes used). 
 95      """ 
 96   
 97      # We are going to store ordered sequence names and the sequences themselves 
 98      order = [] 
 99      sequences = {} 
100   
101      with open(filename) as f: 
102          for line in f: 
103              # Title line for sequences in fasta files start with > 
104              if line.startswith('>'): 
105                  name = line[1:].rstrip('\n') 
106                  name = name.replace('_', ' ') 
107                  order.append(name) 
108                  sequences[name] = '' 
109              elif len(order) > 0: 
110                  # If we have seen a title but not in this line 
111                  # add the contents of the line to the currently named sequence 
112                  # Note: * = chain ending character in fasta so is removed 
113                  sequences[name] += line.rstrip('\n').rstrip('*') 
114   
115      return order, sequences
