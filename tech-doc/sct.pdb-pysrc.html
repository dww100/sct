124      """ 
125      Parse an atom record line from a PDB or PSF file via schema. 
126      Based on the PDB parsing code written by Alisue (lambdalisue@hashnote.net): 
127      https://gist.github.com/lambdalisue/7209305 
128       
129      Schema format: 
130   
131          1) record name 
132          2) start index 
133          3) end index or None for single character 
134          4) processing function or None for `string.strip` 
135   
136          so a truncated (but valid) schema for a PDB ATOM record cold be: 
137   
138          ATOM_RECORD_SCHEMA = ( 
139          ...    ('record', 0,  6,   None),   # 0,  record name 
140          ...    ('name', 6, 11,   int),      # 1,  serial 
141          ) 
142   
143      @type  line:   string 
144      @param line:   Line from a text file. 
145      @type  schema: list 
146      @param schema: List of 4-tuples. The format of the tuples is given above. 
147      @rtype:        dictionary 
148      @return:       Dictionary with keys specifiec as the record name in the  
149                     schema. 
150      """ 
151       
152      vals = {} 
153      for record, start, end, fn in schema: 
154          if end is None: 
155              end = start + 1 
156          if fn is None: 
157              fn = string.strip 
158          try: 
159              vals[record] = fn(line[start:end]) 
160          except ValueError, e: 
161              print "Error: File %s contains a line which could not be parsed according to the schema." % filename 
162              print "Python error message: %s" % e 
163              print "Line contents:", line 
164              raise IOError("Error parsing %s, ignoring this pdb file." % filename) 
165      return vals
